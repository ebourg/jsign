042    /**

044     * 
045     * This class is not thread safe, read access must be synchronized.

047     * @see <a href="http://msdn.microsoft.com/en-us/library/windows/hardware/gg463119.aspx">Microsoft PE and COFF Specification </a>

054        /** The position of the PE header in the file */
055        private long peHeaderOffset;
056    
057        private File file;
058        private ExtendedRandomAccessFile raf;
059    
060        public PEFile(File file) throws IOException {
061            this.file = file;
062            raf = new ExtendedRandomAccessFile(file, "rw");
063            
064            // DOS Header
065            
066            byte[] buffer = new byte[2];
067            raf.read(buffer);
068            
069            if (!Arrays.equals(buffer, "MZ".getBytes())) {
070                throw new IOException("DOS header signature not found");
071            }
072            
073            raf.seek(0x3C);
074            peHeaderOffset = raf.readDWord();
075            
076            // PE Header
077            
078            raf.seek(peHeaderOffset);
079            
080            buffer = new byte[4];
081            raf.read(buffer);
082            if (!Arrays.equals(buffer, new byte[] { 'P', 'E', 0, 0})) {
083                throw new IOException("PE signature not found as expected at offset 0x" + Long.toHexString(peHeaderOffset));
084            }
085        }
086

088            raf.close();
089        }

381        /**
382         * Compute the checksum of the image file.
383         */
384        public long computeChecksum() {
385            try {
386                long checksumOffset = peHeaderOffset + 88;
387    
388                long checksum = 0;
389                long top = (long) Math.pow(2, 32);
390                long length = raf.length();
391    
392                raf.seek(0);
393                for (long i = 0; i < length / 4; i++) {
394                    if (i == checksumOffset / 4) {
395                        raf.skipBytes(4);
396                        continue;
397                    }
398        
399                    long ch1 = raf.read();
400                    long ch2 = raf.read();
401                    long ch3 = raf.read();
402                    long ch4 = raf.read();
403        
404                    long dword = ch1 + (ch2 << 8) + (ch3 << 16) + (ch4 << 24);
405        
406                    checksum = (checksum & 0xffffffffL) + dword + (checksum >> 32);
407                    
408                    if (checksum > top) {
409                        checksum = (checksum & 0xffffffffL) + (checksum >> 32);
410                    }
411                }
412    
413                checksum = (checksum & 0xffff) + (checksum >> 16);
414                checksum = checksum + (checksum >> 16);
415                checksum = checksum & 0xffff;
416                checksum += length;
417    
418                return checksum;
419            } catch (IOException e) {
420                throw new RuntimeException(e);
421            }
422        }
423        
424        public void updateChecksum() {
425            try {
426                ByteBuffer buffer = ByteBuffer.allocate(4);
427                buffer.order(ByteOrder.LITTLE_ENDIAN);
428                buffer.putInt((int) computeChecksum());
429                
430                raf.seek(peHeaderOffset + 88);
431                raf.write(buffer.array());
432            } catch (IOException e) {
433                throw new RuntimeException(e);
434            }
435        }

525        /**
526         * Writes the data directory of the specified type. The data is appended
527         * at the end of the file. If a previous entry was already present it's
528         * left as is in the file and is simply dereferenced from the table. 
529         * 
530         * @param type
531         * @param data
532         * @throws IOException
533         */
534        public void writeDataDirectory(DataDirectoryType type, byte[] data) throws IOException {
535            // todo overwrite an existing entry at the end of the file
536            
537            // append the data directory at the end of the file
538            long offset = raf.length();        
539            
540            raf.seek(offset);
541            raf.write(data);
542            
543            // add the entry in the data directory table
544            DataDirectory entry = getDataDirectory(type);
545            entry.write(offset, data.length);
546            
547            updateChecksum();
548        }

550        /**
551         * Returns the authenticode signatures on the file.
552         */
553        public List<CMSSignedData> getSignatures() {
554            List<CMSSignedData> signatures = new ArrayList<CMSSignedData>();
555            DataDirectory certificateTable = getDataDirectory(DataDirectoryType.CERTIFICATE_TABLE);
556            
557            if (certificateTable != null && certificateTable.getVirtualAddress() != 0 && certificateTable.getSize() != 0) {
558                long position = certificateTable.getVirtualAddress();
559                
560                try {
561                    raf.seek(position);
562                    int size = (int) raf.readDWord();
563                    int revision = raf.readWord();
564                    int type = raf.readWord();
565    
566                    if (revision == 0x0200 && type == 2) {
567                        // PKCS#7 SignedData structure
568                        byte[] buffer = new byte[size - 8];
569                        raf.read(buffer);
570    
571                        CMSSignedData signedData = new CMSSignedData((CMSProcessable) null, ContentInfo.getInstance(buffer));
572                        signatures.add(signedData);
573                    }
574                    
575                    // todo read the remaining signatures
576                } catch (Exception e) {
577                    e.printStackTrace();
578                }
579            }
580            
581            return signatures;
582        }

594        public void printInfo(PrintWriter out) {
595            out.println("PE File");
596            out.println("  Name:          " + file.getName());
597            out.println("  Size:          " + file.length());
598            out.println("  Last Modified: " + new Date(file.lastModified()));
599            out.println();
600            
601            out.println("PE Header");
602            out.println("  Machine:                    " + getMachineType());
603            out.println("  Number of sections:         " + getNumberOfSections());
604            out.println("  Timestamp:                  " + getTimeDateStamp());
605            out.println("  Pointer to symbol table:    0x" + Long.toHexString(getPointerToSymbolTable()));
606            out.println("  Number of symbols:          " + getNumberOfSymbols());
607            out.println("  Size of optional header:    " + getSizeOfOptionalHeader());
608            out.println("  Characteristics:            0x" + Long.toBinaryString(getCharacteristics()));
609            out.println();
610            
611            out.println("Optional Header");
612            PEFormat format = getFormat();
613            out.println("  PE Format:                  0x" + Integer.toHexString(format.value) + " (" + format.label + ")");
614            out.println("  Linker version:             " + getMajorLinkerVersion() + "." + getMinorLinkerVersion());
615            out.println("  Size of code:               " + getSizeOfCode());
616            out.println("  Size of initialized data:   " + getSizeOfInitializedData());
617            out.println("  Size of uninitialized data: " + getSizeOfUninitializedData());
618            out.println("  Address of entry point:     0x" + Long.toHexString(getAddressOfEntryPoint()));
619            out.println("  Base of code:               0x" + Long.toHexString(getBaseOfCode()));
620            if (PEFormat.PE32.equals(getFormat())) {
621                out.println("  Base of data:               0x" + Long.toHexString(getBaseOfData()));
622            }
623            out.println("  Image base:                 0x" + Long.toHexString(getImageBase()));
624            out.println("  Section alignment:          " + getSectionAlignment());
625            out.println("  File alignment:             " + getFileAlignment());
626            out.println("  Operating system version:   " + getMajorOperatingSystemVersion() + "." + getMinorOperatingSystemVersion());
627            out.println("  Image version:              " + getMajorImageVersion() + "." + getMinorImageVersion());
628            out.println("  Subsystem version:          " + getMajorSubsystemVersion() + "." + getMinorSubsystemVersion());
629            out.println("  Size of image:              " + getSizeOfImage());
630            out.println("  Size of headers:            " + getSizeOfHeaders());
631            out.println("  Checksum:                   0x" + Long.toHexString(getCheckSum()));
632            out.println("  Checksum (computed):        0x" + Long.toHexString(computeChecksum()));
633            out.println("  Subsystem:                  " + getSubsystem());
634            out.println("  DLL characteristics:        0x" + Long.toBinaryString(getDllCharacteristics()));
635            out.println("  Size of stack reserve:      " + getSizeOfStackReserve());
636            out.println("  Size of stack commit:       " + getSizeOfStackCommit());
637            out.println("  Size of heap reserve:       " + getSizeOfHeapReserve());
638            out.println("  Size of heap commit:        " + getSizeOfHeapCommit());
639            out.println("  Number of RVA and sizes:    " + getNumberOfRvaAndSizes());
640            out.println();
641            
642            out.println("Data Directory");
643            for (DataDirectoryType type : DataDirectoryType.values()) {
644                DataDirectory entry = getDataDirectory(type);
645                if (entry != null && entry.getVirtualAddress() != 0) {
646                    out.printf("  %-30s 0x%08x %8d bytes\n", type, entry.getVirtualAddress(), entry.getSize());
647                }
648            }
649            out.println();
650            
651            int sectionTableOffset = getDataDirectoryOffset() + 8 * getNumberOfRvaAndSizes(); 
652            out.println("Sections");
653            out.println("      Name     Virtual Size  Virtual Address  Raw Data Size  Raw Data Ptr  Characteristics");
654            for (int i = 0; i < getNumberOfSections(); i++) {
655                Section section = new Section(this, sectionTableOffset + 40 * i);
656                out.printf("  #%d  %-8s     %8d       0x%08x       %8d    0x%08x  %s\n", i + 1, section.getName(), section.getVirtualSize(), section.getVirtualAddress(), section.getSizeOfRawData(), section.getPointerToRawData(), section.getCharacteristics());
657            }
658            out.println();
659            
660            List<CMSSignedData> signatures = getSignatures();
661            if (!signatures.isEmpty()) {
662                out.println("Signatures");
663                for (CMSSignedData signedData : signatures) {
664                    SignerInformation signerInformation = (SignerInformation) signedData.getSignerInfos().getSigners().iterator().next();
665                    X509CertificateHolder certificate = (X509CertificateHolder) signedData.getCertificates().getMatches(signerInformation.getSID()).iterator().next();
666                    
667                    String commonName = certificate.getSubject().getRDNs(X509ObjectIdentifiers.commonName)[0].getFirst().getValue().toString();
668                    
669                    Attribute counterSignature = null;
670                    if (signerInformation.getUnsignedAttributes() != null) {
671                        counterSignature = signerInformation.getUnsignedAttributes().get(PKCSObjectIdentifiers.pkcs_9_at_counterSignature);
672                    }
673                    
674                    out.println("  " + commonName + "   " + (counterSignature != null ? "(timestamped)" : ""));
675                }
676            }
677        }

679        /**
680         * Compute the digest of the file. The checksum field, the certificate
681         * directory table entry and the certificate table are excluded from
682         * the digest.
683         * 
684         * todo Handle files > 2GB
685         */
686        private byte[] computeDigest(MessageDigest digest) throws IOException {
687            int checksumLocation = (int) peHeaderOffset + 88;
688            
689            DataDirectory certificateTable = getDataDirectory(DataDirectoryType.CERTIFICATE_TABLE);
690            
691            // load the file in a buffer
692            byte[] signedContent = new byte[(int) raf.length()];
693            raf.seek(0);
694            raf.read(signedContent);
695            
696            // digest from the beginning to the checksum field (excluded)
697            digest.update(signedContent, 0, checksumLocation);
698            
699            // skip the checksum field
700            int position = checksumLocation + 4;
701            
702            // digest from the end of the checksum field to the beginning of the certificate table entry
703            int certificateTableOffset = getDataDirectoryOffset() + 8 * DataDirectoryType.CERTIFICATE_TABLE.ordinal();
704            digest.update(signedContent, position, certificateTableOffset - position);
705            
706            // skip the certificate entry
707            position = certificateTableOffset + 8;
708            
709            // todo digest the sections in ascending address order
710            
711            // digest from the end of the certificate table entry to the beginning of the certificate table
712            if (certificateTable != null && certificateTable.getVirtualAddress() != 0) {
713                digest.update(signedContent, position, (int) certificateTable.getVirtualAddress() - position);
714                position = (int) certificateTable.getVirtualAddress() + certificateTable.getSize();
715            }
716    
717            // digest from the end of the certificate table to the end of the file
718            digest.update(signedContent, position, (int) raf.length() - position);
719            
720            return digest.digest();
721        }
722        
723        /**
724         * Compute the checksum of the file using the specified digest algorithm.
725         * 
726         * @param algorithm the digest algorithm, typically SHA1
727         */
728        public byte[] computeDigest(String algorithm) throws IOException {
729            try {
730                return computeDigest(MessageDigest.getInstance(algorithm));
731            } catch (NoSuchAlgorithmException e) {
732                throw new RuntimeException(e);
733            }
734        }

736        /**
737         * Increase the size of the file up to a size that is a multiple of the specified value.
738         * 
739         * @param multiple
740         */
741        public void pad(int multiple) throws IOException {
742            long padding = multiple - raf.length() % multiple;
743            raf.seek(raf.length());
744            for (int i = 0; i < padding; i++) {
745                raf.writeByte(0);
746            }
747        }
