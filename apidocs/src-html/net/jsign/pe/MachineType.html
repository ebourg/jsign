017    package net.jsign.pe;
018    
019    /**
020     * Target architecture of an executable file.
021     * 
022     * @author Emmanuel Bourg
023     * @since 1.0
024     */
025    public enum MachineType {
026    
027        /** Executable assumed to be applicable to any machine type */
028        UNKNOWN(0x0),
029    
030        /** Matsushita AM33 */
031        AM33(0x1d3),
032    
033        /** x64 */
034        AMD64(0x8664),
035    
036        /** ARM little endian */
037        ARM(0x1c0),
038    
039        /** ARMv7 (or higher) Thumb mode only */
040        ARMV7(0x1c4),
041    
042        /** EFI byte code */
043        EBC(0xebc),
044    
045        /** Intel 386 or later processors and compatible processors */
046        I386(0x14c),
047    
048        /** Intel Itanium processor family */
049        IA64(0x200),
050    
051        /** Mitsubishi M32R little endian */
052        M32R(0x9041),
053    
054        /** MIPS16 */
055        MIPS16(0x266),
056    
057        /** MIPS with FPU */
058        MIPSFPU(0x366),
059    
060        /** MIPS16 with FPU */
061        MIPSFPU16(0x466),
062    
063        /** Power PC little endian */
064        POWERPC(0x1f0),
065    
066        /** Power PC with floating point support */
067        POWERPCFP(0x1f1),
068    
069        /** MIPS little endian */
070        R4000(0x166),
071    
072        /** Hitachi SH3 */
073        SH3(0x1a2),
074    
075        /** Hitachi SH3 DSP */
076        SH3DSP(0x1a3),
077    
078        /** Hitachi SH4 */
079        SH4(0x1a6),
080    
081        /** Hitachi SH5 */
082        SH5(0x1a8),
083    
084        /** ARM or Thumb (interworking) */
085        THUMB(0x1c2),
086    
087        /** MIPS little-endian WCE v2 */
088        WCEMIPSV2(0x169); 
089    
090        private final int value;
091    
092        MachineType(int value) {
093            this.value = value;
094        }
095    
096        static MachineType valueOf(int value) {
097            for (MachineType format : values()) {
098                if (format.value == value) {
099                    return format;
100                }
101            }
102            
103            return null;
104        }
105    }
