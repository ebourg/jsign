258    /**
259     * Sign the specified executable file.
260     *
261     * @throws Exception
262     */
263    public void sign(PEFile file) throws Exception {
264        // pad the file on a 8 byte boundary
265        // todo only if there was no previous certificate table
266        file.pad(8);
267        
268        if (replace) {
269            DataDirectory certificateTable = file.getDataDirectory(DataDirectoryType.CERTIFICATE_TABLE);
270            if (certificateTable != null && !certificateTable.isTrailing()) {
271                // erase the previous signature
272                certificateTable.erase();
273                certificateTable.write(0, 0);
274            }
275        }
276        
277        // compute the signature
278        CMSSignedData sigData = createSignature(file);
279        
280        // timestamping
281        if (timestamping) {
282            Timestamper ts = timestamper;
283            if (ts == null) {
284                ts = Timestamper.create(tsmode);
285            }
286            if (tsaurlOverride != null) {
287                ts.setURLs(tsaurlOverride);
288            }
289            if (timestampingRetries != -1) {
290                ts.setRetries(timestampingRetries);
291            }
292            if (timestampingRetryWait != -1) {
293                ts.setRetryWait(timestampingRetryWait);
294            }
295            sigData = ts.timestamp(digestAlgorithm, sigData);
296        }
297        
298        List<CMSSignedData> signatures = file.getSignatures();
299        if (!signatures.isEmpty() && !replace) {
300            // append the nested signature
301            sigData = addNestedSignature(signatures.get(0), sigData);
302        }
303
304        CertificateTableEntry entry = new CertificateTableEntry(sigData);
305        
306        file.writeDataDirectory(DataDirectoryType.CERTIFICATE_TABLE, entry.toBytes());
307        file.close();
308    }
309
310    private CMSSignedData addNestedSignature(CMSSignedData primary, CMSSignedData secondary) throws CMSException {
311        SignerInformation signerInformation = primary.getSignerInfos().getSigners().iterator().next();
312        
313        AttributeTable unsignedAttributes = signerInformation.getUnsignedAttributes();
314        if (unsignedAttributes == null) {
315            unsignedAttributes = new AttributeTable(new DERSet());
316        }
317        Attribute nestedSignaturesAttribute = unsignedAttributes.get(AuthenticodeObjectIdentifiers.SPC_NESTED_SIGNATURE_OBJID);
318        if (nestedSignaturesAttribute == null) {
319            // first nested signature
320            unsignedAttributes = unsignedAttributes.add(AuthenticodeObjectIdentifiers.SPC_NESTED_SIGNATURE_OBJID, secondary.toASN1Structure());
321        } else {
322            // append the signature to the previous nested signatures
323            ASN1EncodableVector nestedSignatures = new ASN1EncodableVector();
324            for (ASN1Encodable nestedSignature : nestedSignaturesAttribute.getAttrValues()) {
325                nestedSignatures.add(nestedSignature);
326            }
327            nestedSignatures.add(secondary.toASN1Structure());
328            
329            ASN1EncodableVector attributes = unsignedAttributes.remove(AuthenticodeObjectIdentifiers.SPC_NESTED_SIGNATURE_OBJID).toASN1EncodableVector();
330            attributes.add(new Attribute(AuthenticodeObjectIdentifiers.SPC_NESTED_SIGNATURE_OBJID, new DERSet(nestedSignatures)));
331            
332            unsignedAttributes = new AttributeTable(attributes);
333        }
334        
335        signerInformation = SignerInformation.replaceUnsignedAttributes(signerInformation, unsignedAttributes);
336        return CMSSignedData.replaceSigners(primary, new SignerInformationStore(signerInformation));
337    }

339    private CMSSignedData createSignature(PEFile file) throws IOException, CMSException, OperatorCreationException, CertificateEncodingException {
340        byte[] sha = file.computeDigest(digestAlgorithm);
341        
342        AlgorithmIdentifier algorithmIdentifier = new AlgorithmIdentifier(digestAlgorithm.oid, DERNull.INSTANCE);
343        DigestInfo digestInfo = new DigestInfo(algorithmIdentifier, sha);
344        SpcAttributeTypeAndOptionalValue data = new SpcAttributeTypeAndOptionalValue(AuthenticodeObjectIdentifiers.SPC_PE_IMAGE_DATA_OBJID, new SpcPeImageData());
345        SpcIndirectDataContent spcIndirectDataContent = new SpcIndirectDataContent(data, digestInfo);
346
347        // create content signer
348        final String sigAlg;
349        if (signatureAlgorithm == null) {
350            sigAlg = digestAlgorithm + "with" + privateKey.getAlgorithm();
351        } else {
352            sigAlg = signatureAlgorithm;
353        }
354        JcaContentSignerBuilder contentSignerBuilder = new JcaContentSignerBuilder(sigAlg);
355        if (signatureProvider != null) {
356            contentSignerBuilder.setProvider(signatureProvider);
357        }
358        ContentSigner shaSigner = contentSignerBuilder.build(privateKey);
359
360        DigestCalculatorProvider digestCalculatorProvider = new JcaDigestCalculatorProviderBuilder().build();
361        
362        // prepare the authenticated attributes
363        CMSAttributeTableGenerator attributeTableGenerator = new DefaultSignedAttributeTableGenerator(createAuthenticatedAttributes());
364        
365        // fetch the signing certificate
366        X509CertificateHolder certificate = new JcaX509CertificateHolder((X509Certificate) chain[0]);
367        
368        // prepare the signerInfo with the extra authenticated attributes
369        SignerInfoGeneratorBuilder signerInfoGeneratorBuilder = new SignerInfoGeneratorBuilder(digestCalculatorProvider);
370        signerInfoGeneratorBuilder.setSignedAttributeGenerator(attributeTableGenerator);
371        SignerInfoGenerator signerInfoGenerator = signerInfoGeneratorBuilder.build(shaSigner, certificate);
372        
373        AuthenticodeSignedDataGenerator generator = new AuthenticodeSignedDataGenerator();
374        generator.addCertificates(new JcaCertStore(removeRoot(chain)));
375        generator.addSignerInfoGenerator(signerInfoGenerator);
376        
377        return generator.generate(AuthenticodeObjectIdentifiers.SPC_INDIRECT_DATA_OBJID, spcIndirectDataContent);
378    }

380    /**
381     * Remove the root certificate from the chain, unless the chain consists in a single self signed certificate.
382     */
383    private List<Certificate> removeRoot(Certificate[] certificates) {
384        List<Certificate> list = new ArrayList<>();
385        
386        if (certificates.length == 1) {
387            list.add(certificates[0]);
388        } else {
389            for (Certificate certificate : certificates) {
390                if (!isSelfSigned((X509Certificate) certificate)) {
391                    list.add(certificate);
392                }
393            }
394        }
395        
396        return list;
397    }
398
399    private boolean isSelfSigned(X509Certificate certificate) {
400        return certificate.getSubjectDN().equals(certificate.getIssuerDN());
401    }
402
403    /**
404     * Creates the authenticated attributes for the SignerInfo section of the signature.
405     */
406    private AttributeTable createAuthenticatedAttributes() {
407        List<Attribute> attributes = new ArrayList<>();
408        
409        SpcStatementType spcStatementType = new SpcStatementType(AuthenticodeObjectIdentifiers.SPC_INDIVIDUAL_SP_KEY_PURPOSE_OBJID);
410        attributes.add(new Attribute(AuthenticodeObjectIdentifiers.SPC_STATEMENT_TYPE_OBJID, new DERSet(spcStatementType)));
411        
412        if (programName != null || programURL != null) {
413            SpcSpOpusInfo spcSpOpusInfo = new SpcSpOpusInfo(programName, programURL);
414            attributes.add(new Attribute(AuthenticodeObjectIdentifiers.SPC_SP_OPUS_INFO_OBJID, new DERSet(spcSpOpusInfo)));
415        }
416        
417        return new AttributeTable(new DERSet(attributes.toArray(new ASN1Encodable[attributes.size()])));
418    }
